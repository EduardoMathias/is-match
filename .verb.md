### Create matchers:

**from a string:**

```js
var isMatch = matcher('a')

isMatch('a');
//=> true

isMatch('b');
//=> false
```

**from a glob pattern:**

```js
var isMatch = matcher('*')
isMatch('a'); //=> true

var isMatch = matcher('!b')
isMatch('a'); //=> true

var isMatch = matcher('!b')
isMatch('b'); //=> false
```

**from an array of glob patterns:**

```js
var isMatch = matcher(['b'])
isMatch('a'); //=> false

var isMatch = matcher(['b', 'a'])
isMatch('a'); //=> true

var isMatch = matcher(['b', 'c', '*'])
isMatch('a'); //=> true
```

**from a regex:**

```js
var isMatch = matcher(/a/);

isMatch('a');
//=> true

isMatch('b');
//=> false
```

**from a function:**

```js
var isMatch = matcher(function  (val) {
  return val === 'a';
});
isMatch('a');
//=> true

isMatch('b');
//=> false
```
